    <script>
        // --- SCRIPT SECTION ---
        let currentSlide = 1;
        const totalSlides = 5;

        const prevBtn = document.getElementById('prevBtn');
        const nextBtn = document.getElementById('nextBtn');
        const slideCounter = document.getElementById('slide-counter');

        function showSlide(slideNumber) {
            for (let i = 1; i <= totalSlides; i++) {
                document.getElementById(`slide-${i}`).classList.add('hidden');
            }
            document.getElementById(`slide-${slideNumber}`).classList.remove('hidden');
            
            slideCounter.textContent = `${slideNumber} / ${totalSlides}`;
            
            prevBtn.disabled = slideNumber === 1;
            nextBtn.disabled = slideNumber === totalSlides;
            prevBtn.classList.toggle('opacity-50', slideNumber === 1);
            nextBtn.classList.toggle('opacity-50', slideNumber === totalSlides);
        }

        nextBtn.addEventListener('click', () => {
            if (currentSlide < totalSlides) {
                currentSlide++;
                showSlide(currentSlide);
            }
        });

        prevBtn.addEventListener('click', () => {
            if (currentSlide > 1) {
                currentSlide--;
                showSlide(currentSlide);
            }
        });

        const demoNameInput = document.getElementById('demoName');
        const demoBtn = document.getElementById('demoBtn');
        const demoOutput = document.getElementById('demoOutput');

        demoBtn.addEventListener('click', () => {
            const name = demoNameInput.value.trim() || "Anonymous";
            let message = '';
            if (name.toLowerCase() === "james nelson") {
                message = `Hey, it's the awesome AI Director, James Nelson!`;
            } else {
                const currentHour = new Date().getHours();
                let greeting = '';
                if (currentHour >= 5 && currentHour < 12) greeting = "Good morning";
                else if (currentHour >= 12 && currentHour < 18) greeting = "Good afternoon";
                else greeting = "Good evening";
                message = `${greeting}, ${name}! Welcome to the demo.`;
            }
            demoOutput.innerHTML = `<p class="font-semibold text-white">${message}</p>`;
        });
        
        showSlide(currentSlide);
    </script>
